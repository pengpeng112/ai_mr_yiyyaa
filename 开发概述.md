pythonimport cx_Oracle

def fetch_records(conn, dept_list: list[str], query_date: str) -> list[dict]:
    if not dept_list:
        dept_filter, params = "1=1", {"query_date": query_date}
    else:
        placeholders = ",".join([f":d{i}" for i in range(len(dept_list))])
        dept_filter = f"a.科室 IN ({placeholders})"
        params = {f"d{i}": d for i, d in enumerate(dept_list)}
        params["query_date"] = query_date

    sql = f"""
        SELECT a.患者ID, a.姓名, a.性别, a.年龄, a.科室, a.诊断,
               b.病历标题时间, b.记录内容 AS 病程记录内容,
               c.创建时间, c.护理单类型, c.病情观察及护理措施,
               c.体温, c.心率脉搏, c.呼吸, c.血压, c.意识神志
        FROM jhemr.v_zybr a
        LEFT JOIN jhemr.v_bcjl b ON a.患者ID = b.患者ID AND a.次数 = b.次数
        LEFT JOIN ydhl.v_hljl c
          ON c.患者ID = b.患者ID || '_' || b.次数
         AND TO_CHAR(b.病历标题时间, 'yyyy-mm-dd') = TO_CHAR(c.创建时间, 'yyyy-mm-dd')
        WHERE {dept_filter} AND TO_CHAR(b.病历标题时间, 'yyyy-mm-dd') = :query_date
        ORDER BY a.患者ID, b.病历标题时间
    """
    cursor = conn.cursor()
    cursor.execute(sql, params)
    columns = [d[0] for d in cursor.description]
    return [dict(zip(columns, row)) for row in cursor.fetchall()]

8.2 调用 Dify Workflow（Blocking模式）
pythonimport requests

def push_to_dify(mr_text: str, config: dict, patient_id: str) -> dict:
    url = f"{config['base_url']}/workflows/run"
    headers = {"Authorization": f"Bearer {config['api_key']}", "Content-Type": "application/json"}
    payload = {
        "inputs": {config["workflow_input_variable"]: mr_text},
        "response_mode": "blocking",
        "user": config.get("user_identifier", f"auto-{patient_id}")
    }
    try:
        resp = requests.post(url, headers=headers, json=payload,
                             timeout=config.get("timeout_seconds", 90))
        resp.raise_for_status()
        data = resp.json()
        return {
            "status": "success",
            "workflow_run_id": data.get("workflow_run_id"),
            "task_id": data.get("task_id"),
            "result": data.get("data", {}).get("outputs", {}),
            "elapsed_ms": int(data.get("data", {}).get("elapsed_time", 0) * 1000)
        }
    except requests.exceptions.Timeout:
        return {"status": "failed", "error": "请求超时"}
    except requests.exceptions.HTTPError:
        return {"status": "failed", "error": f"HTTP {resp.status_code}: {resp.text}"}
    except Exception as e:
        return {"status": "failed", "error": str(e)}

        )

        if result.get("inconsistency"):
            send_notification(patient_id, result, config["notify"])

        time.sleep(config.get("interval_ms", 500) / 1000)

密码加密：Oracle 密码和 Dify API Key 必须 AES-256 加密存储，密钥通过环境变量 SECRET_KEY 注入，不写入镜像。
患者隐私：日志中患者姓名可配置脱敏；推送给 Dify 的文本遵循院方数据合规策略。
限流防护：批量推送加入 interval_ms 间隔，避免打爆 Dify 或 Oracle 连接池。
失败重试：对 status=failed 支持手动重推或自动重试（最多3次，指数退避）。
日志保留：推送日志建议保留至少180天，支持研究数据回溯。
Dify 超时：Blocking 模式100秒后可能被中断，timeout_seconds 建议设为 90，或改用 streaming 模式处理 SSE 流。
内网离线：所有依赖（pip 包、Oracle Instant Client）需打包进镜像，EulerOS 服务器无需联网。
数据备份：定期备份 ./data/ 目录（SQLite + 配置文件），可用 cron + rsync 或脚本定时打包。
平台兼容：Windows 开发时使用 docker buildx --platform linux/amd64，确保与 EulerOS x64 服务器架构匹配。
